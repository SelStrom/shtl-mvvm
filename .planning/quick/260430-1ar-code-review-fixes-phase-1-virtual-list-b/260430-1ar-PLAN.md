**3. B-2 + B-7: Переписать `OnElementAdded` с использованием флага `_isFixedLayout`.**
Удалить тавтологию `_vmList.GetItemHeight(0) == _vmList.GetItemHeight(0)` и реализовать сохранение fixed-path при добавлении элементов одной высоты.

Логика:
- Если `_isFixedLayout && newItemHeight == _fixedItemHeight` (с эпсилоном 0.001f) — вызвать `_layoutCalculator.Rebuild(count, _fixedItemHeight)` (это переинициализирует prefix-sum, НО сохранит fixed-flag, обеспечив O(1) GetItemOffset/GetItemHeight).
- Иначе — добавленный элемент отличается по высоте либо был variable-режим: вызвать `_layoutCalculator.InsertAt(index, count, GetHeightProvider())` и установить `_isFixedLayout = false`.

Пример (адаптируй под существующий стиль файла):
```csharp
private void OnElementAdded(int index, TViewModel element)
{
    var count = _vmList.Count;
    _layoutCalculator.SetSpacing(_scrollRect.Spacing);

    var newHeight = _vmList.GetItemHeight(index);
    if (_isFixedLayout && Math.Abs(newHeight - _fixedItemHeight) < 0.001f)
    {
        // O(N) rebuild prefix-sum, но fixed-mode сохраняется → O(1) индексация остаётся.
        _layoutCalculator.Rebuild(count, _fixedItemHeight);
    }
    else
    {
        _layoutCalculator.InsertAt(index, count, GetHeightProvider());
        _isFixedLayout = false;
    }

    _scrollRect.SetContentSize(_layoutCalculator.TotalHeight);

    if (index < _currentRange.FirstIndex)
    {
        _scrollRect.ScrollPosition += _vmList.GetItemHeight(index);
    }

    ShiftActiveViewIndices(index, 1);
    UpdateVisibleRange();
}
```

Логика:
- Если `_isFixedLayout` — после удаления все оставшиеся элементы по-прежнему той же высоты, поэтому `_layoutCalculator.Rebuild(count, _fixedItemHeight)`.

private void OnElementRemoved(int index, TViewModel element)
{
    var heightOfRemoved = _layoutCalculator.GetItemHeight(index);

    if (_activeViews.TryGetValue(index, out var view))
    {
        _recyclingPool.Release(view);
        _activeViews.Remove(index);
    }

    var count = _vmList.Count;
    _layoutCalculator.SetSpacing(_scrollRect.Spacing);

    if (_isFixedLayout)
    {
        _layoutCalculator.Rebuild(count, _fixedItemHeight);
    }
    else
    {
        _layoutCalculator.RemoveAt(index, count, GetHeightProvider());
    }

    _scrollRect.SetContentSize(_layoutCalculator.TotalHeight);

    if (index < _currentRange.FirstIndex)
    {
        _scrollRect.ScrollPosition -= heightOfRemoved;
    }

    ShiftActiveViewIndices(index + 1, -1);
    UpdateVisibleRange();
}

**5. Обновить `RebuildLayout` для установки флагов `_isFixedLayout` / `_fixedItemHeight`** (после определения `isFixed`):
```csharp
private void RebuildLayout(int count)
{
    _layoutCalculator.SetSpacing(_scrollRect.Spacing);

    if (count == 0)
    {
        _layoutCalculator.Rebuild(0, 1f);
        _isFixedLayout = false;
        _fixedItemHeight = 0f;
        return;
    }

    var firstHeight = _vmList.GetItemHeight(0);
    var isFixed = true;
    for (var i = 1; i < count; i++)
    {
        if (Math.Abs(_vmList.GetItemHeight(i) - firstHeight) > 0.001f)
        {
            isFixed = false;
            break;
        }
    }

    if (isFixed)
    {
        _layoutCalculator.Rebuild(count, firstHeight);
        _isFixedLayout = true;
        _fixedItemHeight = firstHeight;
    }
    else
    {
        _layoutCalculator.Rebuild(count, GetHeightProvider());
        _isFixedLayout = false;
        _fixedItemHeight = 0f;
    }
}
```
